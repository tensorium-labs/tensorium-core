# Phase 9A.6 Bridge Relayer + Launch Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build auto bridge relayer (TXM ↔ wTXM Optimism), deploy contracts to OP mainnet, update bridge.tensoriumlabs.com to Phase 9A Beta live page.

**Architecture:** Node.js ESM relayer service with two independent polling loops: deposit-watcher (Tensorium MC → Optimism mint) and withdrawal-watcher (Optimism event → TXM release via txmwallet subprocess). Bridge page is static HTML deployed to VPS at /var/www/bridge/index.html.

**Tech Stack:** Node.js ESM, ethers v6, txmwallet binary (/usr/local/bin/txmwallet), Hardhat, Optimism mainnet (chainId 10), pm2

import { readFileSync, writeFileSync, renameSync, existsSync } from "fs";
import { join, dirname } from "path";
import { fileURLToPath } from "url";

const __dirname = dirname(fileURLToPath(import.meta.url));
const STATE_PATH = join(__dirname, "relayer-state.json");
const TMP_PATH = STATE_PATH + ".tmp";

const DEFAULT = {
  processedUtxos: {},
  processedWithdrawals: {},
  lastProcessedBlock: 0,
};

export function readState() {
  if (!existsSync(STATE_PATH)) return structuredClone(DEFAULT);
  try {
    return JSON.parse(readFileSync(STATE_PATH, "utf8"));
  } catch {
    console.error("[state] corrupt state file, resetting");
    return structuredClone(DEFAULT);
  }
}

export function writeState(state) {
  writeFileSync(TMP_PATH, JSON.stringify(state, null, 2));
  renameSync(TMP_PATH, STATE_PATH);
}

## Task 7: op-client.js

**Files:**
- Create: `tensorium-bridge-relayer/op-client.js`

- [ ] **Step 1: Write op-client.js**

```javascript
import { ethers } from "ethers";

const CONTROLLER_ABI = [
  "function mintFromTensoriumDeposit(bytes32 bridgeEventId, bytes32 tensoriumTxid, address recipient, uint256 amount) external",
  "function maxPerTx() view returns (uint256)",
  "function paused() view returns (bool)",
  "event WithdrawalRequested(bytes32 indexed bridgeEventId, address indexed requester, string tensoriumAddress, uint256 amount)",
];

// 1 TXM = 1e8 atoms; 1 wTXM = 1e18 wei
// atoms → wei: multiply by 1e10
export function atomsToWei(atoms) {
  return BigInt(atoms) * 10n ** 10n;
}

// wei → atoms: divide by 1e10
export function weiToAtoms(wei) {
  return BigInt(wei) / 10n ** 10n;
}

export function makeBridgeEventId(txid, outputIndex) {
  return ethers.keccak256(
    ethers.AbiCoder.defaultAbiCoder().encode(
      ["bytes32", "uint32"],
      [ethers.zeroPadBytes(ethers.toBeHex(txid), 32), outputIndex]
    )
  );
}

export function createOpClient() {
  const provider = new ethers.JsonRpcProvider(process.env.OP_RPC_URL);
  const wallet = new ethers.Wallet(process.env.OPERATOR_PRIVATE_KEY, provider);
  const controller = new ethers.Contract(
    process.env.CONTROLLER_ADDRESS,
    CONTROLLER_ABI,
    wallet
  );
  return { provider, wallet, controller };
}

export async function mintFromDeposit(controller, bridgeEventId, txid, recipient, amountAtoms) {
  const amountWei = atomsToWei(amountAtoms);
  const txidBytes32 = ethers.zeroPadBytes(ethers.toBeHex("0x" + txid), 32);
  const tx = await controller.mintFromTensoriumDeposit(
    bridgeEventId,
    txidBytes32,
    recipient,
    amountWei
  );
  return tx.wait();
}

// Returns WithdrawalRequested events from fromBlock to toBlock
export async function getWithdrawalEvents(provider, controller, fromBlock, toBlock) {
  const filter = controller.filters.WithdrawalRequested();
  return controller.queryFilter(filter, fromBlock, toBlock);
}
```

- [ ] **Step 1: Write deposit-watcher.js**

```javascript
import { readFileSync, existsSync } from "fs";
import { join, dirname } from "path";
import { fileURLToPath } from "url";
import { getBlockCount, getUtxos } from "./txm-client.js";
import { createOpClient, mintFromDeposit, makeBridgeEventId } from "./op-client.js";
import { readState, writeState } from "./state.js";

const __dirname = dirname(fileURLToPath(import.meta.url));
const PENDING_DIR = join(__dirname, "deposits", "pending");
const MIN_CONFS = parseInt(process.env.MIN_CONFIRMATIONS || "6");
const CUSTODY_ADDR = process.env.CUSTODY_ADDRESS;

function pendingFile(txid) {
  return join(PENDING_DIR, `${txid}.json`);
}

function loadPending(txid) {
  const path = pendingFile(txid);
  if (!existsSync(path)) return null;
  try {
    return JSON.parse(readFileSync(path, "utf8"));
  } catch {
    console.error(`[deposit] corrupt pending file for ${txid}`);
    return null;
  }
}

export async function runDepositWatcher() {
  const { controller } = createOpClient();

  console.log("[deposit] watching custody address:", CUSTODY_ADDR);
  console.log("[deposit] min confirmations:", MIN_CONFS);

  while (true) {
    try {
      await checkDeposits(controller);
    } catch (err) {
      console.error("[deposit] loop error:", err.message);
    }
    await new Promise((r) =>
      setTimeout(r, parseInt(process.env.POLL_INTERVAL_DEPOSIT_MS || "60000"))
    );
  }
}

async function checkDeposits(controller) {
  const { tip_height, utxos } = await getUtxos(CUSTODY_ADDR);
  const state = readState();
  let changed = false;

  for (const utxo of utxos) {
    const key = `${utxo.txid}:${utxo.output_index}`;
    if (state.processedUtxos[key]) continue;

    const confs = tip_height - utxo.created_height;
    if (confs < MIN_CONFS) {
      console.log(`[deposit] UTXO ${key} has ${confs}/${MIN_CONFS} confs, waiting`);
      continue;
    }

    const pending = loadPending(utxo.txid);
    if (!pending) {
      console.warn(`[deposit] UTXO ${key} confirmed but no pending file — needs operator mapping`);
      continue;
    }

    const bridgeEventId = makeBridgeEventId(utxo.txid, utxo.output_index);
    console.log(`[deposit] minting ${utxo.value_atoms} atoms → ${pending.recipient}, eventId=${bridgeEventId}`);

    try {
      const receipt = await mintFromDeposit(
        controller,
        bridgeEventId,
        utxo.txid,
        pending.recipient,
        utxo.value_atoms
      );
      console.log(`[deposit] minted, evmTx=${receipt.hash}`);
      state.processedUtxos[key] = {
        mintTx: receipt.hash,
        recipient: pending.recipient,
        atoms: utxo.value_atoms,
        ts: new Date().toISOString(),
      };
      changed = true;
    } catch (err) {
      console.error(`[deposit] mint failed for ${key}:`, err.message);
    }
  }

  if (changed) writeState(state);
}
```

- Create: `tensorium-bridge-relayer/withdrawal-watcher.js`

- [ ] **Step 1: Write withdrawal-watcher.js**

```javascript
import { execSync } from "child_process";
import { writeFileSync, unlinkSync, existsSync } from "fs";
import { join, dirname } from "path";
import { fileURLToPath } from "url";
import { createOpClient, getWithdrawalEvents, weiToAtoms } from "./op-client.js";
import { readState, writeState } from "./state.js";

const __dirname = dirname(fileURLToPath(import.meta.url));
const WALLET_PATH = process.env.CUSTODY_WALLET_PATH;
const WALLET_PASS = process.env.CUSTODY_WALLET_PASSPHRASE;
const MC_STATE   = process.env.TENSORIUM_MC_STATE;
const MC_RPC_LOCAL = process.env.TENSORIUM_MC_RPC_LOCAL || "127.0.0.1:33332";

function releaseTxm(tensoriumAddress, atoms) {
  const tmpTx = join(__dirname, `txm-release-${Date.now()}.json`);
  const env = {
    ...process.env,
    TENSORIUM_WALLET: WALLET_PATH,
    TENSORIUM_WALLET_PASSPHRASE: WALLET_PASS,
    TENSORIUM_STATE: MC_STATE,
  };
  try {
    execSync(`txmwallet send ${tensoriumAddress} ${atoms} ${tmpTx}`, { env });
    const result = execSync(`txmwallet broadcast ${tmpTx} ${MC_RPC_LOCAL}`, {
      env,
      encoding: "utf8",
    });
    return result.trim();
  } finally {
    if (existsSync(tmpTx)) unlinkSync(tmpTx);
  }
}

export async function runWithdrawalWatcher() {
  const { provider, controller } = createOpClient();

  const state = readState();
  if (state.lastProcessedBlock === 0) {
    state.lastProcessedBlock = await provider.getBlockNumber();
    writeState(state);
    console.log("[withdrawal] initialized lastProcessedBlock:", state.lastProcessedBlock);
  }

  console.log("[withdrawal] watching from block:", state.lastProcessedBlock);

  while (true) {
    try {
      await checkWithdrawals(provider, controller);
    } catch (err) {
      console.error("[withdrawal] loop error:", err.message);
    }
    await new Promise((r) =>
      setTimeout(r, parseInt(process.env.POLL_INTERVAL_WITHDRAWAL_MS || "30000"))
    );
  }
}

async function checkWithdrawals(provider, controller) {
  const state = readState();
  const currentBlock = await provider.getBlockNumber();
  const fromBlock = state.lastProcessedBlock + 1;

  if (fromBlock > currentBlock) return;

  const events = await getWithdrawalEvents(provider, controller, fromBlock, currentBlock);
  let changed = false;

  for (const event of events) {
    const { bridgeEventId, tensoriumAddress, amount } = event.args;
    if (state.processedWithdrawals[bridgeEventId]) continue;

    const atoms = weiToAtoms(amount);
    console.log(`[withdrawal] releasing ${atoms} atoms to ${tensoriumAddress}, eventId=${bridgeEventId}`);

    try {
      const result = releaseTxm(tensoriumAddress, atoms.toString());
      console.log(`[withdrawal] released, txmwallet output: ${result}`);
      state.processedWithdrawals[bridgeEventId] = {
        tensoriumAddr: tensoriumAddress,
        atoms: atoms.toString(),
        evmTx: event.transactionHash,
        ts: new Date().toISOString(),
      };
      changed = true;
    } catch (err) {
      console.error(`[withdrawal] release failed for ${bridgeEventId}:`, err.message);
    }
  }

  state.lastProcessedBlock = currentBlock;
  writeState(state);
}
```
